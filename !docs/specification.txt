+ acc
----------------------
+ id              [id]
+ username        unique<string>
+ password        unique<hash>
+ firstname       string
+ lastname        string
+ telephone       string|null
+ address         string|null
+ acc_type        enum( 'adm', 'em', 'st' )
+ created         date|null
+ active          date|null
+ deleted         date|null
### employee ###
+ em_title        -> acc_em_title|null
+ em_biography    string|null
+ em_photo        -> file|null
+ em_website_url  string|null
+ em_cabinet      string|null
### student ###
+ st_index        string|null
+ st_semester     number|null

+ acc_em_title
----------------------
+ id              [id]
+ title           unique<string>   # nastavno zvanje:   redovni profesor, vanredni profesor, docent, asistent, saradnik u nastavi
                                   # nenastavno zvanje: istraživač, laboratorijski inženjer, laboratorijski tehničar

+ acc_group
----------------------
+ id              [id]
+ name            string|null
+ type            string
+ acc             list< -> acc >
+ created         date
+ deleted         date|null



+ news
----------------------
+ id              [id]
+ title           string
+ text            string|null
+ news_cat        -> news_cat
+ acc             -> acc
+ scope           list< ->general|->crs >
+ files           list< ->file >|null
+ created         date
+ posted          date
+ expired         date             # kada obavestenje treba da posivi
+ deleted         date|null

+ news_cat
----------------------
+ id              [id]
+ category        unique<string>   # takmičenje, konferencija, praksa, posao, predmet
+ created         date
+ deleted         date|null



+ crs
----------------------
+ id              [id]
+ code            unique<string>
+ name            string
+ plan            list< >
  + department      enum( 'si', 'rti', 'os' )
  + semester        number
  + optional        bool
+ class_hh        string
+ espb            number
+ lecturer        list< ->acc >
+ goal            string
+ outcome         string
+ page            list< >
  + type            enum( 'info', 'pred', 'vezbe', 'lab', 'domaci', 'proj', 'ispit' )
  + text            string|null
  + files           list< ->file >|null
  + enabled         date|null
+ created         date
+ deleted         date|null

+ crs_group
----------------------
+ id              [id]
+ crs             -> crs
+ department      enum( 'si', 'rti', 'os' )
+ semester        number
+ lecturer        list< ->acc >
+ student         -> acc_group
+ tm_schoolyear   number
+ tm_weekday      number
+ tm_timebeg      time
+ tm_timeend      time

+ deleted         date|null

    + file            -> file|null
+ created         date
+ readonly        date|null
+ completed       date|null
